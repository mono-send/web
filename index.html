send-email.ts
import { Monosend } from 'monosend';

const monosend = new Monosend('mono_123456789');

await monosend.emails.send({
  from: 'onboarding@monosend',
  to: 'user@gmail.com',
  subject: 'Hello World',
  html: '<p>Congrats on sending your <strong>first email</strong>!</p>'
});
